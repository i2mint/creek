creek.scrap.async_utils¶

Utils to deal with async iteration

Making singledispatch work:

``` from typing import Generator, Iterator, Iterable, AsyncIterable, AsyncIterator from functools import singledispatch, partial

from typing import Protocol, runtime_checkable

@runtime_checkable class IterableType(Protocol):

def __iter__(self):

pass

@runtime_checkable class CursorFunc(Protocol):

def __call__(self):

pass

@singledispatch def to_iterator(x: IterableType):

return iter(x)

will_never_happen = object()

@to_iterator.register def _(x: CursorFunc):

return iter(x, will_never_happen)

assert list(to_iterator([1,2,3])) == [1, 2, 3]

f = partial(next, iter([1,2,3])) assert list(to_iterator(f)) == [1, 2, 3] ```

async for i in async_iterable:

yield i

t = [i async for i in my_aiter(ticker(3, 0.2))] assert t == [0, 1, 2]
