    name: str,
    tp: type = Any,
    doc: Optional[str] = None,
    aka: Optional[Union[str, Iterable[str]]] = None,
):
    """Like `typing.NewType` with some extras (`__doc__` and `_aka` attributes, etc.)
    """

    new_tp = NewType(name, tp)
    if doc is not None:
        new_tp.__doc__ = doc
    if aka is not None:
        new_tp._aka = aka
    return new_tp
